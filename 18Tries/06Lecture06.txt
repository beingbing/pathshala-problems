string ans = ""
void findUniquePrefix(string &s, node* cur, int pos) {
    if (cur->strings == 1) ans = s.substr(0, pos), return;
    findUniquePrefix(s, cur->children[s[pos]-'a'], pos+1);
}

for finding TC = O(nl)
