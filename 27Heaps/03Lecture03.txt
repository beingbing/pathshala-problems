~~ building a max heap - 2 ~~

if we have n elements then our last non-null node will be n/2.

void maxHeapify(int i);

// buildMaxHeap
for (i{n/2}; i>=1) {
    maxHeapify(i);
}

void maxHeapify(int i) {
    if (i>n/2) return;
    int max = i;
    if (a[2*i] > a[max]) max = 2*i;
    if (2*i+1<=n && a[2*i+1] > a[max]) max=2*i+1;
    if (max != i) swap(a[i], a[max]), maxHeapify(i);
}
TC = O(n)
