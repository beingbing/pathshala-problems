~~ reverse linked list (recursively) ~~

recursive steps -
- delink the head from remainder.
- newHead = make recursive call on remainder
- link back
- return new head.

Node* rev(Node* p) {
    if (!p) return p;
    if (!(p->next)) return p;
    Node* tmp = p->next;
    p->next = NULL;
    Node* head = rev(tmp);
    tmp->next = p;
    return head;
}
TC = O(n)
SC = O(n)
